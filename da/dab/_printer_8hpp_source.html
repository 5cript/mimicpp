188 class PrintFn
189 {
190 public:
191 template <print_iterator OutIter, typename T>
192 OutIter operator ()(
193 OutIter out,
194 T&& value
195 ) const
196 {
197 static_assert(
198 requires(const priority_tag<4> tag)
199 {
200 { print(out, std::forward<T>(value), tag) } -> std::convertible_to<OutIter>;
201 },
202 "The given type is not printable. ");
203
204 return print(
205 out,
206 std::forward<T>(value),
207 priority_tag<4>{});
208 }

210 template <typename T>
211 StringT operator ()(T&& value) const
212 {
213 StringStreamT stream{};
214 operator()(
215 std::ostreambuf_iterator{stream},
216 std::forward<T>(value));
217 return std::move(stream).str();
218 }
219 };

221 template <typename OutIter, std::ranges::forward_range Range>
222 OutIter print(
223 OutIter out,
224 Range&& range,
225 const priority_tag<2>
226 )
227 {
228 out = format::format_to(out, "{{ ");
229 auto iter = std::ranges::begin(range);
230 if (const auto end = std::ranges::end(range);
231 iter != end)
232 {
233 constexpr PrintFn print{};
234 out = print(out, *iter++);
235
236 for (; iter != end; ++iter)
237 {
238 out = print(
239 format::format_to(out, ", "),
240 *iter);
241 }
242 }
243
244 return format::format_to(out, " }}");
245 }
