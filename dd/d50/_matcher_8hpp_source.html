40 template <typename Predicate, typename... AdditionalArgs>
41 requires std::is_move_constructible_v<Predicate>
42 && (... && std::is_move_constructible_v<AdditionalArgs>)
43 class PredicateMatcher
44 {
45 public:
46 [[nodiscard]]
47 explicit constexpr PredicateMatcher(
48 Predicate predicate,
49 StringT fmt,
50 std::tuple<AdditionalArgs...> additionalArgs = std::tuple{}
51 ) noexcept(std::is_nothrow_move_constructible_v<Predicate>
52 && (... && std::is_nothrow_move_constructible_v<AdditionalArgs>))
53 : m_Predicate{std::move(predicate)},
54 m_FormatString{std::move(fmt)},
55 m_AdditionalArgs{std::move(additionalArgs)}
56 {
57 }
58
59 template <typename T>
60 requires std::predicate<const Predicate&, T&, const AdditionalArgs&...>
61 [[nodiscard]]
62 constexpr bool matches(
63 T& target
64 ) const noexcept(std::is_nothrow_invocable_v<const Predicate&, T&, const AdditionalArgs&...>)
65 {
66 return std::apply(
67 [&, this](auto&... additionalArgs)
68 {
69 return std::invoke(
70 m_Predicate,
71 target,
72 additionalArgs...);
73 },
74 m_AdditionalArgs);
75 }

77 template <typename T>
78 [[nodiscard]]
79 constexpr StringT describe(T& target) const
80 {
81 return std::apply(
82 [&, this](auto&... additionalArgs)
83 {
84 // std::make_format_args requires lvalue-refs, so let's transform rvalue-refs to const lvalue-refs
85 constexpr auto makeLvalue = [](auto&& val) noexcept -> const auto& { return val; };
86 return format::vformat(
87 m_FormatString,
88 format::make_format_args(
89 makeLvalue(mimicpp::print(target)),
90 makeLvalue(mimicpp::print(additionalArgs))...));
91 },
92 m_AdditionalArgs);
93 }
94
95 [[nodiscard]]
96 constexpr auto operator !() const &
97 requires std::is_copy_constructible_v<Predicate>
98 && (... && std::is_copy_constructible_v<AdditionalArgs>)
99 {
100 return make_inverted(
101 m_Predicate,
102 m_FormatString,
103 std::move(m_AdditionalArgs));
104 }
105
106 [[nodiscard]]
107 constexpr auto operator !() &&
108 {
109 return make_inverted(
110 std::move(m_Predicate),
111 m_FormatString,
112 std::move(m_AdditionalArgs));
113 }
114
115 private:
116 [[no_unique_address]] Predicate m_Predicate;
117 StringT m_FormatString;
118 mutable std::tuple<AdditionalArgs...> m_AdditionalArgs{};
119
120 template <typename Fn>
121 [[nodiscard]]
122 static constexpr auto make_inverted(Fn&& fn, const StringViewT fmt, std::tuple<AdditionalArgs...> tuple)
123 {
124 using NotFnT = decltype(std::not_fn(std::forward<Fn>(fn)));
125 return PredicateMatcher<NotFnT, AdditionalArgs...>{
126 std::not_fn(std::forward<Fn>(fn)),
127 format::format("not ({})", fmt),
128 std::move(tuple)
129 };
130 }
