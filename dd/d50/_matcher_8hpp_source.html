129 template <typename Predicate, typename... AdditionalArgs>
130 requires std::is_move_constructible_v<Predicate>
131 && (... && std::is_move_constructible_v<AdditionalArgs>)
132 class PredicateMatcher
133 {
134 public:
135 [[nodiscard]]
136 explicit constexpr PredicateMatcher(
137 Predicate predicate,
138 StringT fmt,
139 StringT invertedFmt,
140 std::tuple<AdditionalArgs...> additionalArgs = std::tuple{}
141 ) noexcept(std::is_nothrow_move_constructible_v<Predicate>
142 && (... && std::is_nothrow_move_constructible_v<AdditionalArgs>))
143 : m_Predicate{std::move(predicate)},
144 m_FormatString{std::move(fmt)},
145 m_InvertedFormatString{std::move(invertedFmt)},
146 m_AdditionalArgs{std::move(additionalArgs)}
147 {
148 }
149
150 template <typename T>
151 requires std::predicate<const Predicate&, T&, const AdditionalArgs&...>
152 [[nodiscard]]
153 constexpr bool matches(
154 T& target
155 ) const noexcept(std::is_nothrow_invocable_v<const Predicate&, T&, const AdditionalArgs&...>)
156 {
157 return std::apply(
158 [&, this](auto&... additionalArgs)
159 {
160 return std::invoke(
161 m_Predicate,
162 target,
163 additionalArgs...);
164 },
165 m_AdditionalArgs);
166 }
167
168 [[nodiscard]]
169 constexpr StringT describe() const
170 {
171 return std::apply(
172 [&, this](auto&... additionalArgs)
173 {
174 // std::make_format_args requires lvalue-refs, so let's transform rvalue-refs to const lvalue-refs
175 constexpr auto makeLvalue = [](auto&& val) noexcept -> const auto& { return val; };
176 return format::vformat(
177 m_FormatString,
178 format::make_format_args(
179 makeLvalue(mimicpp::print(additionalArgs))...));
180 },
181 m_AdditionalArgs);
182 }
183
184 [[nodiscard]]
185 constexpr auto operator !() const &
186 requires std::is_copy_constructible_v<Predicate>
187 && (... && std::is_copy_constructible_v<AdditionalArgs>)
188 {
189 return make_inverted(
190 m_Predicate,
191 m_InvertedFormatString,
192 m_FormatString,
193 std::move(m_AdditionalArgs));
194 }
195
196 [[nodiscard]]
197 constexpr auto operator !() &&
198 {
199 return make_inverted(
200 std::move(m_Predicate),
201 std::move(m_InvertedFormatString),
202 std::move(m_FormatString),
203 std::move(m_AdditionalArgs));
204 }
205
206 private:
207 [[no_unique_address]] Predicate m_Predicate;
208 StringT m_FormatString;
209 StringT m_InvertedFormatString;
210 mutable std::tuple<AdditionalArgs...> m_AdditionalArgs{};
211
212 template <typename Fn>
213 [[nodiscard]]
214 static constexpr auto make_inverted(
215 Fn&& fn,
216 StringT fmt,
217 StringT invertedFmt,
218 std::tuple<AdditionalArgs...> tuple
219 )
220 {
221 using NotFnT = decltype(std::not_fn(std::forward<Fn>(fn)));
222 return PredicateMatcher<NotFnT, AdditionalArgs...>{
223 std::not_fn(std::forward<Fn>(fn)),
224 std::move(fmt),
225 std::move(invertedFmt),
226 std::move(tuple)
227 };
228 }
229 };
