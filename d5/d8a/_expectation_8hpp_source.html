27 template <typename Return, typename... Params, typename Signature>
28 std::optional<MatchReport> make_match_report(
29 const call::Info<Return, Params...>& call,
30 const Expectation<Signature>& expectation
31 ) noexcept
32 {
33 try
34 {
35 return expectation.matches(call);
36 }
37 catch (...)
38 {
39 report_unhandled_exception(
40 make_call_report(call),
41 expectation.report(),
42 std::current_exception());
43 }
44
45 return std::nullopt;
46 }

48 template <typename Signature>
49 constexpr auto pick_best_match(std::vector<std::tuple<Expectation<Signature>&, MatchReport>>& matches)
50 {
51 constexpr auto ratings = [](const auto& el) noexcept -> const auto& {
52 return std::get<state_applicable>(
53 std::get<MatchReport>(el).controlReport)
54 .sequenceRatings;
55 };
56
57 auto best = std::ranges::begin(matches);
58 for (auto iter = best + 1;
59 iter != std::ranges::end(matches);
60 ++iter)
61 {
62 if (!sequence::detail::has_better_rating(
63 ratings(*best),
64 ratings(*iter)))
65 {
66 best = iter;
67 }
68 }
69
70 return best;
71 }

160 void push(std::shared_ptr<ExpectationT> expectation)
161 {
162 const std::scoped_lock lock{m_ExpectationsMx};
163
164 assert(
165 std::ranges::find(m_Expectations, expectation) == std::ranges::end(m_Expectations)
166 && "Expectation already belongs to this storage.");
167
168 m_Expectations.emplace_back(std::move(expectation));
169 }
170
171 void remove(std::shared_ptr<ExpectationT> expectation)
172 {
173 const std::scoped_lock lock{m_ExpectationsMx};
174
175 auto iter = std::ranges::find(m_Expectations, expectation);
176 assert(iter != std::ranges::end(m_Expectations) && "Expectation does not belong to this storage.");
177 m_Expectations.erase(iter);
178
179 if (!expectation->is_satisfied())
180 {
181 detail::report_unfulfilled_expectation(
182 expectation->report());
183 }
184 }
185
186 [[nodiscard]]
187 ReturnT handle_call(const CallInfoT& call)
188 {
189 std::vector<std::tuple<ExpectationT&, MatchReport>> matches{};
190 std::vector<MatchReport> noMatches{};
191 std::vector<MatchReport> inapplicableMatches{};
192
193 for (const std::scoped_lock lock{m_ExpectationsMx};
194 auto& exp : m_Expectations | std::views::reverse)
195 {
196 if (std::optional matchReport = detail::make_match_report(call, *exp))
197 {
198 switch (evaluate_match_report(*matchReport))
199 {
200 using enum MatchResult;
201 case none:
202 noMatches.emplace_back(*std::move(matchReport));
203 break;
204 case inapplicable:
205 inapplicableMatches.emplace_back(*std::move(matchReport));
206 break;
207 case full:
208 matches.emplace_back(*exp, *std::move(matchReport));
209 break;
210 // GCOVR_EXCL_START
211 default:
212 unreachable();
213 // GCOVR_EXCL_STOP
214 }
215 }
216 }
217
218 if (!std::ranges::empty(matches))
219 {
220 auto&& [exp, report] = *detail::pick_best_match(matches);
221 detail::report_full_match(
222 make_call_report(call),
223 std::move(report));
224 exp.consume(call);
225 return exp.finalize_call(call);
226 }
227
228 if (!std::ranges::empty(inapplicableMatches))
229 {
230 detail::report_inapplicable_matches(
231 make_call_report(call),
232 std::move(inapplicableMatches));
233 }
234
235 detail::report_no_matches(
236 make_call_report(call),
237 std::move(noMatches));
238 }
239
240 private:
241 std::vector<std::shared_ptr<ExpectationT>> m_Expectations{};
242 std::mutex m_ExpectationsMx{};
243 };

245 template <typename T, typename Signature>
246 concept expectation_policy_for = std::is_move_constructible_v<T>
247 && std::is_destructible_v<T>
248 && std::same_as<T, std::remove_cvref_t<T>>
249 && requires(T& policy, const call::info_for_signature_t<Signature>& info)
250 {
251 { std::as_const(policy).is_satisfied() } noexcept -> std::convertible_to<bool>;
252 { std::as_const(policy).matches(info) } -> std::convertible_to<bool>;
253 { std::as_const(policy).describe() } -> std::convertible_to<std::optional<StringT>>;
254 { policy.consume(info) };
255 };
256
257 template <typename T, typename Signature>
258 concept finalize_policy_for = std::is_move_constructible_v<T>
259 && std::is_destructible_v<T>
260 && std::same_as<T, std::remove_cvref_t<T>>
261 && requires(T& policy, const call::info_for_signature_t<Signature>& info)
262 {
263 { policy.finalize_call(info) } -> std::convertible_to<signature_return_type_t<Signature>>;
264 };
265
266 template <typename T>
267 concept control_policy = std::is_move_constructible_v<T>
268 && std::is_destructible_v<T>
269 && std::same_as<T, std::remove_cvref_t<T>>
270 && requires(T& policy)
271 {
272 { std::as_const(policy).is_satisfied() } noexcept -> std::convertible_to<bool>;
273 { std::as_const(policy).state() } -> std::convertible_to<control_state_t>;
274 policy.consume();
275 };

283 class BasicExpectation final
284 : public Expectation<Signature>
285 {
286 public:
287 using ControlPolicyT = ControlPolicy;
288 using FinalizerT = FinalizePolicy;

293 template <typename ControlPolicyArg, typename FinalizerArg, typename... PolicyArgs>
294 requires std::constructible_from<ControlPolicyT, ControlPolicyArg>
295 && std::constructible_from<FinalizerT, FinalizerArg>
296 && std::constructible_from<PolicyListT, PolicyArgs...>
297 constexpr explicit BasicExpectation(
298 const std::source_location& sourceLocation,
299 ControlPolicyArg&& controlArg,
300 FinalizerArg&& finalizerArg,
301 PolicyArgs&&... args
302 ) noexcept(
303 std::is_nothrow_constructible_v<ControlPolicyT, ControlPolicyArg>
304 && std::is_nothrow_constructible_v<FinalizerT, FinalizerArg>
305 && (std::is_nothrow_constructible_v<Policies, PolicyArgs> && ...))
306 : m_SourceLocation{sourceLocation},
307 m_ControlPolicy{std::forward<ControlPolicyArg>(controlArg)},
308 m_Policies{std::forward<PolicyArgs>(args)...},
309 m_Finalizer{std::forward<FinalizerArg>(finalizerArg)}
310 {
311 }
312
313 [[nodiscard]]
314 ExpectationReport report() const override
315 {
316 return ExpectationReport{
317 .sourceLocation = m_SourceLocation,
318 .finalizerDescription = std::nullopt,
319 .timesDescription = std::invoke(
320 [this]
321 {
322 StringStreamT ss{};
323 std::visit(
324 std::bind_front(
325 detail::control_state_printer{},
326 std::ostreambuf_iterator{ss}),
327 m_ControlPolicy.state());
328 return std::move(ss).str();
329 }),
330 .expectationDescriptions = std::apply(
331 [&](const auto&... policies)
332 {
333 return std::vector<std::optional<StringT>>{
334 policies.describe()...
335 };
336 },
337 m_Policies)
338 };
339 }
340
341 [[nodiscard]]
342 constexpr bool is_satisfied() const noexcept override
343 {
344 return m_ControlPolicy.is_satisfied()
345 && std::apply(
346 [](const auto&... policies) noexcept
347 {
348 return (... && policies.is_satisfied());
349 },
350 m_Policies);
351 }
352
353 [[nodiscard]]
354 MatchReport matches(const CallInfoT& call) const override
355 {
356 return MatchReport{
357 .sourceLocation = m_SourceLocation,
358 .finalizeReport = {std::nullopt},
359 .controlReport = m_ControlPolicy.state(),
360 .expectationReports = std::apply(
361 [&](const auto&... policies)
362 {
363 return std::vector<MatchReport::Expectation>{
364 MatchReport::Expectation{
365 .isMatching = policies.matches(call),
366 .description = policies.describe()
367 }...
368 };
369 },
370 m_Policies)
371 };
372 }
373
374 constexpr void consume(const CallInfoT& call) override
375 {
376 m_ControlPolicy.consume();
377 std::apply(
378 [&](auto&... policies) noexcept
379 {
380 (..., policies.consume(call));
381 },
382 m_Policies);
383 }
384
385 [[nodiscard]]
386 constexpr ReturnT finalize_call(const CallInfoT& call) override
387 {
388 return m_Finalizer.finalize_call(call);
389 }
390
391 [[nodiscard]]
392 constexpr const std::source_location& from() const noexcept override
393 {
394 return m_SourceLocation;
395 }
396
397 private:
398 std::source_location m_SourceLocation;
399 ControlPolicyT m_ControlPolicy;
400 PolicyListT m_Policies;
401 [[no_unique_address]] FinalizerT m_Finalizer{};
402 };
