85 public:

102 void push(std::shared_ptr<ExpectationT> expectation)
103 {
104 const std::scoped_lock lock{m_ExpectationsMx};
105
106 assert(
107 std::ranges::find(m_Expectations, expectation) == std::ranges::end(m_Expectations)
108 && "Expectation already belongs to this storage.");
109
110 m_Expectations.emplace_back(std::move(expectation));
111 }
112
113 void remove(std::shared_ptr<ExpectationT> expectation)
114 {
115 const std::scoped_lock lock{m_ExpectationsMx};
116
117 auto iter = std::ranges::find(m_Expectations, expectation);
118 assert(iter != std::ranges::end(m_Expectations) && "Expectation does not belong to this storage.");
119 m_Expectations.erase(iter);
120
121 if (!expectation->is_satisfied())
122 {
123 detail::report_unfulfilled_expectation(
124 expectation->report());
125 }
126 }
127
128 [[nodiscard]]
129 ReturnT handle_call(const CallInfoT& call)
130 {
131 std::vector<MatchReport> noMatches{};
132 std::vector<MatchReport> inapplicableMatches{};
133
134 for (const std::scoped_lock lock{m_ExpectationsMx};
135 auto& exp : m_Expectations | std::views::reverse)
136 {
137 if (std::optional matchReport = detail::make_match_report(call, *exp))
138 {
139 switch (evaluate_match_report(*matchReport))
140 {
141 using enum MatchResult;
142 case none:
143 noMatches.emplace_back(*std::move(matchReport));
144 break;
145 case inapplicable:
146 inapplicableMatches.emplace_back(*std::move(matchReport));
147 break;
148 case full:
149 detail::report_full_match(
150 make_call_report(call),
151 *std::move(matchReport));
152 exp->consume(call);
153 return exp->finalize_call(call);
154
155 // GCOVR_EXCL_START
156 default:
157 unreachable();
158 // GCOVR_EXCL_STOP
159 }
160 }
161 }
162
163 if (!std::ranges::empty(inapplicableMatches))
164 {
165 detail::report_inapplicable_matches(
166 make_call_report(call),
167 std::move(inapplicableMatches));
168 }
169
170 detail::report_no_matches(
171 make_call_report(call),
172 std::move(noMatches));
173 }
174
175 private:
176 std::vector<std::shared_ptr<ExpectationT>> m_Expectations{};
177 std::mutex m_ExpectationsMx{};
178 };

180 template <typename T, typename Signature>
181 concept expectation_policy_for = std::is_move_constructible_v<T>
182 && std::is_destructible_v<T>
183 && std::same_as<T, std::remove_cvref_t<T>>
184 && requires(T& policy, const call::info_for_signature_t<Signature>& info)
185 {
186 { std::as_const(policy).is_satisfied() } noexcept -> std::convertible_to<bool>;
187 { std::as_const(policy).matches(info) } -> std::convertible_to<bool>;
188 { std::as_const(policy).describe() } -> std::convertible_to<std::optional<StringT>>;
189 { policy.consume(info) };
190 };
191
192 template <typename T, typename Signature>
193 concept finalize_policy_for = std::is_move_constructible_v<T>
194 && std::is_destructible_v<T>
195 && std::same_as<T, std::remove_cvref_t<T>>
196 && requires(T& policy, const call::info_for_signature_t<Signature>& info)
197 {
198 { policy.finalize_call(info) } -> std::convertible_to<signature_return_type_t<Signature>>;
199 };

201 template <typename T>
202 concept times_policy = std::is_move_constructible_v<T>
203 && std::is_destructible_v<T>
204 && std::same_as<T, std::remove_cvref_t<T>>
205 && requires(T& policy)
206 {
207 { std::as_const(policy).is_satisfied() } noexcept -> std::convertible_to<bool>;
208 { std::as_const(policy).is_applicable() } noexcept -> std::convertible_to<bool>;
209 { std::as_const(policy).describe_state() } -> std::convertible_to<std::optional<StringT>>;
210 policy.consume();
211 };

213 template <
214 typename Signature,
215 times_policy TimesPolicy,
216 finalize_policy_for<Signature> FinalizePolicy,
217 expectation_policy_for<Signature>... Policies
218 >
219 class BasicExpectation final
220 : public Expectation<Signature>
221 {
222 public:
223 using TimesT = TimesPolicy;
224 using FinalizerT = FinalizePolicy;
225 using PolicyListT = std::tuple<Policies...>;
226 using CallInfoT = call::info_for_signature_t<Signature>;
227 using ReturnT = typename Expectation<Signature>::ReturnT;
228
229 template <typename TimesArg, typename FinalizerArg, typename... PolicyArgs>
230 requires std::constructible_from<TimesT, TimesArg>
231 && std::constructible_from<FinalizerT, FinalizerArg>
232 && std::constructible_from<PolicyListT, PolicyArgs...>
233 constexpr explicit BasicExpectation(
234 const std::source_location& sourceLocation,
235 TimesArg&& timesArg,
236 FinalizerArg&& finalizerArg,
237 PolicyArgs&&... args
238 ) noexcept(
239 std::is_nothrow_constructible_v<TimesT, TimesArg>
240 && std::is_nothrow_constructible_v<FinalizerT, FinalizerArg>
241 && (std::is_nothrow_constructible_v<Policies, PolicyArgs> && ...))
242 : m_SourceLocation{sourceLocation},
243 m_Policies{std::forward<PolicyArgs>(args)...},
244 m_Times{std::forward<TimesArg>(timesArg)},
245 m_Finalizer{std::forward<FinalizerArg>(finalizerArg)}
246 {
247 }
248
249 [[nodiscard]]
250 ExpectationReport report() const override
251 {
252 return ExpectationReport{
253 .sourceLocation = m_SourceLocation,
254 .finalizerDescription = std::nullopt,
255 .timesDescription = m_Times.describe_state(),
256 .expectationDescriptions = std::apply(
257 [&](const auto&... policies)
258 {
259 return std::vector<std::optional<StringT>>{
260 policies.describe()...
261 };
262 },
263 m_Policies)
264 };
265 }
266
267 [[nodiscard]]
268 constexpr bool is_satisfied() const noexcept override
269 {
270 return m_Times.is_satisfied()
271 && std::apply(
272 [](const auto&... policies) noexcept
273 {
274 return (... && policies.is_satisfied());
275 },
276 m_Policies);
277 }
278
279 [[nodiscard]]
280 MatchReport matches(const CallInfoT& call) const override
281 {
282 return MatchReport{
283 .sourceLocation = m_SourceLocation,
284 .finalizeReport = {std::nullopt},
285 .timesReport = MatchReport::Times{
286 .isApplicable = m_Times.is_applicable(),
287 .description = m_Times.describe_state()
288 },
289 .expectationReports = std::apply(
290 [&](const auto&... policies)
291 {
292 return std::vector<MatchReport::Expectation>{
293 MatchReport::Expectation{
294 .isMatching = policies.matches(call),
295 .description = policies.describe()
296 }...
297 };
298 },
299 m_Policies)
300 };
301 }
302
303 constexpr void consume(const CallInfoT& call) override
304 {
305 m_Times.consume();
306 std::apply(
307 [&](auto&... policies) noexcept
308 {
309 (..., policies.consume(call));
310 },
311 m_Policies);
312 }
313
314 [[nodiscard]]
315 constexpr ReturnT finalize_call(const CallInfoT& call) override
316 {
317 return m_Finalizer.finalize_call(call);
318 }
319
320 [[nodiscard]]
321 constexpr const std::source_location& from() const noexcept
322 {
323 return m_SourceLocation;
324 }
325
326 private:
327 std::source_location m_SourceLocation;
328 PolicyListT m_Policies;
329 [[no_unique_address]] TimesT m_Times{};
330 [[no_unique_address]] FinalizerT m_Finalizer{};
331 };
