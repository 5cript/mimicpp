48 template <typename Return, typename... Params, typename Signature>
49 std::optional<call::MatchResultT> make_match_result(
50 const call::Info<Return, Params...>& call,
51 const std::shared_ptr<Expectation<Signature>>& expectation
52 ) noexcept
53 {
54 try

56 return expectation->matches(call);

120 void push(std::shared_ptr<ExpectationT> expectation)
121 {
122 const std::scoped_lock lock{m_ExpectationsMx};
123
124 assert(
125 std::ranges::find(m_Expectations, expectation) == std::ranges::end(m_Expectations)
126 && "Expectation already belongs to this storage.");
127
128 m_Expectations.emplace_back(std::move(expectation));
129 }
130
131 void remove(std::shared_ptr<ExpectationT> expectation) noexcept
132 {
133 const std::scoped_lock lock{m_ExpectationsMx};
134
135 auto iter = std::ranges::find(m_Expectations, expectation);
136 assert(iter != std::ranges::end(m_Expectations) && "Expectation does not belong to this storage.");
137 m_Expectations.erase(iter);
138
139 if (!expectation->is_satisfied())
140 {
141 report_unsatisfied_expectation(std::move(expectation));
142 }
143 }
144
145 [[nodiscard]]
146 ReturnT handle_call(const CallInfoT& call)
147 {
148 static_assert(3 == std::variant_size_v<call::MatchResultT>, "Unexpected MatchResult alternative count.");
149
150 std::vector<call::MatchResult_NoT> noMatches{};
151 std::vector<call::MatchResult_NotApplicableT> exhaustedMatches{};
152
153 for (const std::scoped_lock lock{m_ExpectationsMx};
154 auto& exp : m_Expectations | std::views::reverse)
155 {
156 if (std::optional matchResult = detail::make_match_result(call, exp))
157 {
158 if (auto* match = std::get_if<call::MatchResult_OkT>(&*matchResult))
159 {
160 report_ok(
161 call,
162 std::move(*match));
163 exp->consume(call);
164 return exp->finalize_call(call);
165 }
166
167 if (auto* match = std::get_if<call::MatchResult_NoT>(&*matchResult))
168 {
169 noMatches.emplace_back(std::move(*match));
170 }
171 else
172 {
173 exhaustedMatches.emplace_back(std::get<call::MatchResult_NotApplicableT>(*std::move(matchResult)));
174 }
175 }
176 }
177
178 detail::handle_call_match_fail(
179 call,
180 std::move(noMatches),
181 std::move(exhaustedMatches));
182 }
183
184 private:
185 std::vector<std::shared_ptr<ExpectationT>> m_Expectations{};
186 std::mutex m_ExpectationsMx{};
187 };

189 template <typename T, typename Signature>
190 concept expectation_policy_for = std::is_move_constructible_v<T>
191 && std::is_destructible_v<T>
192 && std::same_as<T, std::remove_cvref_t<T>>
193 && requires(T& policy, const call::info_for_signature_t<Signature>& info)
194 {
195 { std::as_const(policy).is_satisfied() } noexcept -> std::convertible_to<bool>;
196 { std::as_const(policy).matches(info) } -> std::convertible_to<call::SubMatchResult>;
197 { policy.consume(info) };
198 };
199
200 template <typename T, typename Signature>
201 concept finalize_policy_for = std::is_move_constructible_v<T>
202 && std::is_destructible_v<T>
203 && std::same_as<T, std::remove_cvref_t<T>>
204 && requires(T& policy, const call::info_for_signature_t<Signature>& info)
205 {
206 { policy.finalize_call(info) } -> std::convertible_to<signature_return_type_t<Signature>>;
207 };

209 template <typename T>
210 concept times_policy = std::is_move_constructible_v<T>
211 && std::is_destructible_v<T>
212 && std::same_as<T, std::remove_cvref_t<T>>
213 && requires(T& policy)
214 {
215 { std::as_const(policy).is_satisfied() } noexcept -> std::convertible_to<bool>;
216 { std::as_const(policy).is_applicable() } noexcept -> std::convertible_to<bool>;
217 policy.consume();
218 };

220 template <
221 typename Signature,
222 times_policy TimesPolicy,
223 finalize_policy_for<Signature> FinalizePolicy,
224 expectation_policy_for<Signature>... Policies
225 >
226 class BasicExpectation final
227 : public Expectation<Signature>
228 {
229 public:
230 using TimesT = TimesPolicy;
231 using FinalizerT = FinalizePolicy;
232 using PolicyListT = std::tuple<Policies...>;
233 using CallInfoT = call::info_for_signature_t<Signature>;
234 using ReturnT = typename Expectation<Signature>::ReturnT;
235
236 template <typename TimesArg, typename FinalizerArg, typename... PolicyArgs>
237 requires std::constructible_from<TimesT, TimesArg>
238 && std::constructible_from<FinalizerT, FinalizerArg>
239 && std::constructible_from<PolicyListT, PolicyArgs...>
240 constexpr explicit BasicExpectation(
241 TimesArg&& timesArg,
242 FinalizerArg&& finalizerArg,
243 PolicyArgs&&... args
244 ) noexcept(
245 std::is_nothrow_constructible_v<TimesT, TimesArg>
246 && std::is_nothrow_constructible_v<FinalizerT, FinalizerArg>
247 && (std::is_nothrow_constructible_v<Policies, PolicyArgs> && ...))
248 : m_Times{std::forward<TimesArg>(timesArg)},
249 m_Finalizer{std::forward<FinalizerArg>(finalizerArg)},
250 m_Policies{std::forward<PolicyArgs>(args)...}
251 {
252 }
253
254 [[nodiscard]]
255 constexpr bool is_satisfied() const noexcept override
256 {
257 return m_Times.is_satisfied()
258 && std::apply(
259 [](const auto&... policies) noexcept
260 {
261 return (... && policies.is_satisfied());
262 },
263 m_Policies);
264 }
265
266 [[nodiscard]]
267 call::MatchResultT matches(const CallInfoT& call) const override
268 {
269 return call::detail::evaluate_sub_match_results(
270 m_Times.is_applicable(),
271 std::apply(
272 [&](const auto&... policies)
273 {
274 return std::vector<call::SubMatchResult>{
275 policies.matches(call)...
276 };
277 },
278 m_Policies));
279 }
280
281 constexpr void consume(const CallInfoT& call) override
282 {
283 m_Times.consume();
284 std::apply(
285 [&](auto&... policies) noexcept
286 {
287 (..., policies.consume(call));
288 },
289 m_Policies);
290 }
291
292 [[nodiscard]]
293 constexpr ReturnT finalize_call(const CallInfoT& call) override
294 {
295 return m_Finalizer.finalize_call(call);
296 }
297
298 private:
299 [[no_unique_address]] TimesT m_Times{};
300 [[no_unique_address]] FinalizerT m_Finalizer{};
301 PolicyListT m_Policies;
302 };

311 ~ScopedExpectation()

318 }

320 [[nodiscard]]

324 ) noexcept

326 m_Expectation{std::move(expectation)}
