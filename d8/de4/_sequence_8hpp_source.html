112 constexpr bool is_consumable(const SequenceId id) const noexcept
113 {
114 assert(to_underlying(id) < m_Entries.size());
115
116 return m_Current == to_underlying(id);
117 }

119 [[nodiscard]]
120 constexpr bool is_saturated(const SequenceId id) const noexcept
121 {
122 assert(to_underlying(id) < m_Entries.size());
123
124 const auto [amount, counter] = m_Entries[to_underlying(id)];
125 return amount == counter;
126 }
127
128 constexpr void consume(const SequenceId id) noexcept
129 {
130 assert(is_consumable(id));
131
132 if (auto& [amount, counter] = m_Entries[m_Current];
133 amount == ++counter)
134 {
135 ++m_Current;
136 }
137 }
138
139 [[nodiscard]]
140 constexpr SequenceId add(const std::size_t count)
141 {
142 if (count == 0)
143 {
144 throw std::invalid_argument{"Count must be greater than 0."};
145 }
146
147 m_Entries.emplace_back(count, 0);
148 return SequenceId{m_Entries.size() - 1};
149 }

154 std::size_t amount{};
155 std::size_t counter{};
156 };

222 {

226 }
