123 assert(to_underlying(id) < m_Entries.size());

129 constexpr void consume(const SequenceId id) noexcept
130 {
131 assert(is_consumable(id));
132
133 if (auto& [amount, counter] = m_Entries[m_Current];
134 amount == ++counter)
135 {
136 ++m_Current;
137 }
138 }

145 throw std::invalid_argument{"Count must be greater than 0."};

148 m_Entries.emplace_back(count, 0);

150 }

222 constexpr bool is_satisfied() const noexcept
223 {
224 return std::ranges::all_of(
225 m_SequenceInfos,
226 [](const entry& info){ return info.sequence->is_saturated(info.id); });
227 }

229 [[nodiscard]]
230 constexpr bool is_applicable() const noexcept
231 {
232 return std::ranges::all_of(
233 m_SequenceInfos,

235 }

238 StringT describe_state() const
239 {
240 if (is_applicable())
241 {
242 return "applicable: Sequence element expects further matches.";
243 }
244
245 if (is_satisfied())
246 {
247 return "inapplicable: Sequence element is already saturated.";
248 }
249
250 return "inapplicable: Sequence element is not the current element.";
251 }

254 constexpr void consume() noexcept

256 assert(is_applicable());
